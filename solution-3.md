import "@openzeppelin/contracts/access/Ownable.sol";

contract erc20zk is ERC20, ERC20Burnable, Ownable {
    constructor() ERC20("Aditya", "aak") {
        _mint(msg.sender, 100 * 10 ** decimals());
    }

    }
}
